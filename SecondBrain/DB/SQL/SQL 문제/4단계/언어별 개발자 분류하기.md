```sql
SELECT 
    GRADE,
    ID,
    EMAIL
FROM 
    (
        SELECT 
            (
                CASE
                    WHEN ((D.SKILL_CODE & (SELECT sum(code) FROM SKILLCODES WHERE NAME = 'Python')) > 0) 
                        AND ((D.SKILL_CODE & (SELECT sum(code) FROM SKILLCODES WHERE CATEGORY = 'Front End')) > 0)   
                    THEN 'A'
                    
                    WHEN ((D.SKILL_CODE & (SELECT sum(code) FROM SKILLCODES WHERE NAME = 'C#')) > 0) 
                    THEN 'B'
                    
                    WHEN ((D.SKILL_CODE & (SELECT sum(code) FROM SKILLCODES WHERE CATEGORY = 'Front End')) > 0) 
                    THEN 'C'
                    
                    ELSE NULL
                END
            ) AS GRADE,
            D.ID,
            D.EMAIL
        FROM DEVELOPERS AS D
    ) AS SUB
WHERE GRADE IS NOT NULL
ORDER BY GRADE, ID;
```
