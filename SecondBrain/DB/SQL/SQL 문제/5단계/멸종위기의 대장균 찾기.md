```sql
WITH RECURSIVE GENERATIONS AS (
    SELECT 
        ID,
        PARENT_ID,
        1 AS GENERATION
    FROM ECOLI_DATA
    WHERE PARENT_ID IS NULL
    
    UNION ALL
    
    SELECT 
        ED.ID,
        ED.PARENT_ID,
        G.GENERATION + 1
    FROM ECOLI_DATA ED
    INNER JOIN GENERATIONS G ON ED.PARENT_ID = G.ID
)

SELECT
    COUNT(*) AS COUNT,
    GENERATION
FROM GENERATIONS
WHERE ID NOT IN (
    -- 자식이 있는 개체(누군가의 부모인 ID)를 제외
    SELECT DISTINCT PARENT_ID 
    FROM ECOLI_DATA 
    WHERE PARENT_ID IS NOT NULL
)
GROUP BY GENERATION
ORDER BY GENERATION;
```
