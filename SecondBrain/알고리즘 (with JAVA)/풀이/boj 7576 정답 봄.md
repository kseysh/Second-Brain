시간 초과 남
기존 BFS와 다른 방식으로 풀어야하는 활용문제
**핵심 풀이: BFS를 한 방향만으로 탐색하는 것이 아닌, 여러 출발점에서부터 출발할 수도 있다!**

```java
// 시간 초과난 풀이
import java.io.BufferedReader;  
import java.io.IOException;  
import java.io.InputStreamReader;  
import java.util.LinkedList;  
import java.util.Queue;  
import java.util.StringTokenizer;  
  
public class Main {  
  
    public static void main(String[] args) throws IOException {  
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));  
        StringTokenizer st = new StringTokenizer(br.readLine());  
        boolean[][] isVisited;  
        int N = Integer.parseInt(st.nextToken());  
        int M = Integer.parseInt(st.nextToken());  
        int[] dx = {0,1,0,-1};  
        int[] dy = {1,0,-1,0};  
        int[][] tomatoBox = new int[M][N];  
        int[][] dayCount = new int[M][N];  
  
        for (int i = 0; i < M; i++) {  
            st = new StringTokenizer(br.readLine());  
            for (int j = 0; j < N; j++) {  
                tomatoBox[i][j] = Integer.parseInt(st.nextToken());  
            }  
        }  
  
        for (int i = 0; i < M; i++) {  
            for (int j = 0; j < N; j++) {  
                if (tomatoBox[i][j] == 1) {  
                    Queue<int[]> queue = new LinkedList<>();  
                    isVisited = new boolean[M][N];  
                    queue.add(new int[]{i,j});  
                    isVisited[i][j] = true;  
                    dayCount[i][j] = 1;  
                    while (!queue.isEmpty()) {  
                        int[] xy = queue.poll();  
                        for (int k = 0; k < 4; k++) {  
                            int x = xy[0] + dx[k];  
                            int y = xy[1] + dy[k];  
                            if (x >= 0 && y >= 0 && x < M && y < N) {  
                                if (!isVisited[x][y]) {  
                                    if (tomatoBox[x][y] == -1) {  
                                        isVisited[x][y] = true;  
                                    } else {  
                                        queue.add(new int[]{x,y});  
                                        isVisited[x][y] = true;  
                                        if (dayCount[x][y] == 0) {  
                                            dayCount[x][y] = dayCount[xy[0]][xy[1]] + 1;  
                                        } else {  
                                            if (dayCount[x][y] > dayCount[xy[0]][xy[1]] + 1) {  
                                                dayCount[x][y] = dayCount[xy[0]][xy[1]] + 1;  
                                            }  
                                        }  
                                    }  
                                }  
                            }  
                        }  
                    }  
                }  
            }  
        }  
        int result = 0;  
        for (int i = 0; i < M; i++) {  
            for (int j = 0; j < N; j++) {  
                int current = dayCount[i][j];  
                if (current == 0 && tomatoBox[i][j] != -1) {  
                    System.out.println(-1);  
                    return;  
                }  
                if (current > result) {  
                    result = current;  
                }  
            }  
        }  
        System.out.println(result - 1);  
    }  
}
```
