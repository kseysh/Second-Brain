```java
import java.io.*;
import java.util.*;

class Main {

    static class Pos{
        int y, x, d;
        Pos(int y, int x, int d){
            this.y = y;
            this.x = x;
            this.d = d;
        }
    }

    static int[][] map;
    static boolean[][] isVisit;
    static int[] dy = {0, 1, 0, -1};
    static int[] dx = {1, 0, -1, 0};
    static int n, cy, cx;
    static int sharkSize = 2;
    static int ateFishCount = 0;
    static final int BABY_SHARK = 9;



    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        n = Integer.parseInt(br.readLine());

        map = new int[n][n];

        StringTokenizer st;
        for (int i = 0; i < n; i++) {
            st = new StringTokenizer(br.readLine());
            for (int j = 0; j < n; j++) {
                map[i][j] = Integer.parseInt(st.nextToken());
                if(map[i][j] == BABY_SHARK){
                    cy = i;
                    cx = j;
                    map[i][j] = 0;
                }
            }
        }


        int time = 0;
        while(true){
            int eatingTime = eatFish();
            if(eatingTime == -1) break;
            time += eatingTime;
        }

        System.out.println(time);
    }

    private static int eatFish(){
        initIsVisit();
        int ateFishY = 10000;
        int ateFishX = 10000;
        int minDistance = 10000;

        Queue<Pos> q = new ArrayDeque<>();
        q.add(new Pos(cy, cx, 0));
        isVisit[cy][cx] = true;
        while(!q.isEmpty()){
            Pos pos = q.poll();
            if (pos.d >= minDistance) continue;

            for (int d = 0; d < 4; d++) {
                int ny = pos.y + dy[d];
                int nx = pos.x + dx[d];

                if(!inRange(ny, nx) || isVisit[ny][nx]) continue;

                int fishSize = map[ny][nx];
                if(sharkSize < fishSize) continue;

                isVisit[ny][nx] = true;

                q.add(new Pos(ny, nx, pos.d + 1));
                if(fishSize == 0) continue;
                if(sharkSize > fishSize){
                    if(pos.d + 1 > minDistance) continue;
                    if(ny < ateFishY || (ny == ateFishY && nx < ateFishX)){
                        ateFishY = ny;
                        ateFishX = nx;
                        minDistance = pos.d + 1;
                    }
                }
            }
        }

        // eat
        if(minDistance != 10000){
            if(++ateFishCount == sharkSize){
                sharkSize++;
                ateFishCount = 0;
            }
            map[ateFishY][ateFishX] = 0;
            cy = ateFishY;
            cx = ateFishX;
            return minDistance;
        }

        return -1;
    }

    private static boolean inRange(int y, int x){
        return y < n && x < n && y >= 0 && x >= 0;
    }

    private static void initIsVisit(){
        isVisit = new boolean[n][n];
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                isVisit[i][j] = false;
            }
        }
    }
}
```
