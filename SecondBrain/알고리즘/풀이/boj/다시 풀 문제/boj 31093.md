```java
import java.io.*;
import java.util.*;

public class Main {

    private static int[] mixtureCounts = new int['E' + 'E' + 1];

    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

        StringTokenizer st = new StringTokenizer(br.readLine());
        int n = Integer.parseInt(st.nextToken());
        int k = Integer.parseInt(st.nextToken());

        String str = br.readLine();

        Integer maxScore = 0;
        int half = n / 2;
        for (int i = 0; i < half; i++) {
            int value = str.charAt(half - i - 1) + str.charAt(i + half);
            if(value == 'E' + '3' || value == '2' + '2') maxScore += 2;
            else mixtureCounts[value]++;
        }
        StringBuilder sb = new StringBuilder(maxScore.toString()).append('\n');
        while(k-- > 0){
            // 5C2 = 10개의 상황 중 33,33 / EE,EE / EE,E2 / 33,32 빼고 6가지 상황
            if(mixtureCounts['3' + '3'] >= 1 && mixtureCounts['E' + 'E'] >= 1){
                mixtureCounts['3' + '3']--;
                mixtureCounts['E' + 'E']--;
                maxScore += 4;
            }else if(mixtureCounts['3' + '2'] >= 1 && mixtureCounts['E' + '2'] >= 1){
                mixtureCounts['3' + '2']--;
                mixtureCounts['E' + '2']--;
                maxScore += 4;
            }else if(mixtureCounts['3' + '2'] >= 2){
                mixtureCounts['3' + '2']-=2;
                mixtureCounts['3' + '3']++;
                maxScore += 2;
            }else if(mixtureCounts['E' + '2'] >= 2){
                mixtureCounts['E' + '2']-=2;
                mixtureCounts['E' + 'E']++;
                maxScore += 2;
            }else if(mixtureCounts['3' + '3'] >= 1 && mixtureCounts['E' + '2'] >= 1){
                mixtureCounts['3' + '3']--;
                mixtureCounts['E' + '2']--;
                mixtureCounts['3' + '2']++;
                maxScore += 2;
            }else if(mixtureCounts['E' + 'E'] >= 1 && mixtureCounts['3' + '2'] >= 1){
                mixtureCounts['E' + 'E']--;
                mixtureCounts['3' + '2']--;
                mixtureCounts['E' + '2']++;
                maxScore += 2;
            }
            sb.append(maxScore).append('\n');
        }

        bw.write(sb.toString());
        bw.close();
    }
}
```
