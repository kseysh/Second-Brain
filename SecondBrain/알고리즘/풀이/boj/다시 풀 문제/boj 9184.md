## 나의 풀이

```java
import java.io.*;
import java.util.*;

public class Main {
    public static final int MAX_VALUE = 20;
    public static final long EMPTY = Long.MIN_VALUE;
    public static int answer;
    public static int a;
    public static int b;
    public static int c;
    public static long[][][] dp = new long[MAX_VALUE + 1][MAX_VALUE + 1][MAX_VALUE + 1];

    public static long fillDp(int i, int j, int k){
        if(i <= 0 || j <= 0 || k <= 0) return 1;
        if(dp[i][j][k] != EMPTY) return dp[i][j][k];

        if(i < j && j < k){
            dp[i][j][k] = fillDp(i,j,k-1) + fillDp(i,j-1,k-1) - fillDp(i,j-1,k);
        }else{
            dp[i][j][k] = fillDp(i-1,j,k) + fillDp(i-1,j-1,k) + fillDp(i-1,j,k-1) - fillDp(i-1,j-1,k-1);
        }
        return dp[i][j][k];
    }

    public static void main(String[] args) throws IOException{
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

        for (int i = 0; i <= MAX_VALUE; i++) {
            for (int j = 0; j <= MAX_VALUE; j++) {
                for (int k = 0; k <= MAX_VALUE; k++) {
                    dp[i][j][k] = EMPTY;
                }
            }
        }

        while(true){
            StringTokenizer st = new StringTokenizer(br.readLine());
            a = Integer.parseInt(st.nextToken());
            b = Integer.parseInt(st.nextToken());
            c = Integer.parseInt(st.nextToken());
            if(a == -1 && b == -1 && c == -1) break;
            long answer;
            if(a <= 0 || b <= 0 || c <= 0) answer = 1;
            else if(a > 20 || b > 20 || c > 20) answer = fillDp(MAX_VALUE,MAX_VALUE,MAX_VALUE);
            else answer = fillDp(a,b,c);

            StringBuilder sb = new StringBuilder("w(")
                    .append(a).append(", ")
                    .append(b).append(", ")
                    .append(c).append(") = ");
            bw.write(sb.append(answer).append('\n').toString());
        }
        bw.close();
    }
}

```
`(a <= 0 || b <= 0 || c <= 0)`  이 조건이 먼저 들어갔어야 하지만,
`(a > 20 || b > 20 || c > 20)` 이 조건이 먼저 들어가서 w(-1,20,21)의 경우 오류 발생


## 더 깔끔한 코드
```java
import java.io.*;
import java.util.*;

public class Main {
    public static final int MAX = 20;
    public static final long EMPTY = Long.MIN_VALUE;
    public static long[][][] dp = new long[MAX + 1][MAX + 1][MAX + 1];

    public static long w(int a, int b, int c) {
        if (a <= 0 || b <= 0 || c <= 0) return 1;
        if (a > 20 || b > 20 || c > 20) return w(20, 20, 20);
        if (dp[a][b][c] != EMPTY) return dp[a][b][c];

        if (a < b && b < c)
            dp[a][b][c] = w(a, b, c - 1) + w(a, b - 1, c - 1) - w(a, b - 1, c);
        else
            dp[a][b][c] = w(a - 1, b, c) + w(a - 1, b - 1, c) + w(a - 1, b, c - 1) - w(a - 1, b - 1, c - 1);

        return dp[a][b][c];
    }

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        BufferedWriter bw = new BufferedWriter(new OutputStreamWriter(System.out));

        for (int i = 0; i <= MAX; i++)
            for (int j = 0; j <= MAX; j++)
                Arrays.fill(dp[i][j], EMPTY);

        while (true) {
            StringTokenizer st = new StringTokenizer(br.readLine());
            int a = Integer.parseInt(st.nextToken());
            int b = Integer.parseInt(st.nextToken());
            int c = Integer.parseInt(st.nextToken());
            if (a == -1 && b == -1 && c == -1) break;

            bw.write(String.format("w(%d, %d, %d) = %d\n", a, b, c, w(a, b, c)));
        }

        bw.flush();
        bw.close();
    }
}
```
