#include <vector>
#include <iostream>
#define MAX_N 20
using namespace std;

int n;
int zone[MAX_N + 1][MAX_N + 1];

int calculateZone1(int ty, int tx, int ly, int lx){
    int sum = 0;
    int count = 1;
    for(int y = 0; y < ly; y++){
        if(ty <= y) tx--; 
        for(int x = 0; x <= tx; x++){
            sum += zone[y][x];
        }
    }

    return sum;
}

int calculateZone2(int ty, int tx, int ry, int rx){
    int sum = 0;
    int count = 1;
    for(int y = 0; y <= ry; y++){
        if(ty < y) count++; 
        for(int x = tx + count; x < n; x++){
            sum += zone[y][x];
        }
    }

    return sum;
}

int calculateZone3(int ly, int lx, int by, int bx){
    int sum = 0;
    int bound = lx;
    for(int y = ly; y < n; y++){
        for(int x = 0; x < bound; x++){
            sum += zone[y][x];
        }
        if(bound < bx) bound++;
    }
    return sum;
}

int calculateZone4(int by, int bx, int ry, int rx){
    int sum = 0;
    int count = 0;
    for(int y = ry + 1; y < n; y++){
        for(int x = max(rx - count, bx); x < n; x++){
            sum += zone[y][x];
        }
        count++;
    }

    return sum;
}

int main(){
    
    cin >> n;

    int peopleSum = 0;
    for(int i = 0; i < n; i++){
        for(int j = 0; j < n; j++){
            cin >> zone[i][j];
            peopleSum += zone[i][j];
        }
    }
    int count = 0;
    int answer = 100000000;
    for(int x = 0; x < n - 1; x++){
        for(int y = 1; y < n - 1; y++){
            for(int d1 = 1; d1 < n - 1; d1++){
                int ly = y;
                int lx = x;
                int ty = y - d1;
                int tx = x + d1;
                if(ty < 0) continue;
                for(int d2 = 1; d2 < n - 1; d2++){
                    int ry = y + d2 - d1;
                    int rx = x + d2 + d1;
                    int by = y + d2;
                    int bx = x + d2;

                    if(rx >= n) continue;
                    if(by >= n) continue;
                    int zone1 = calculateZone1(ty, tx, ly, lx);
                    int zone2 = calculateZone2(ty, tx, ry, rx);
                    int zone3 = calculateZone3(ly, lx, by, bx);
                    int zone4 = calculateZone4(by, bx, ry, rx);
                    int zone5 = peopleSum - zone1 - zone2 - zone3 - zone4;
                    
                    int maxZone = max(zone1, max(zone2, max(zone3, max(zone4, zone5))));
                    int minZone = min(zone1, min(zone2, min(zone3, min(zone4, zone5))));

                    answer = min(answer, maxZone - minZone);
                }
            }
        }
    }

    cout << answer;

    return 0;
}
