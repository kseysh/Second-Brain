## 나의 풀이
```java
import java.io.*;
import java.util.*;

class Main {

    static class Pos {
        int y, x;

        Pos(int y, int x){
            this.y = y;
            this.x = x;
        }
    }

    static final int MAX_N = 6;
    static final int MAX_LEN = 1 << MAX_N;
    static final int[] dy = {0, 1, 0, -1};
    static final int[] dx = {1, 0, -1, 0};

    static int n, q, len;
    static int[][] grid = new int[MAX_LEN][MAX_LEN];
    static int[][] temp = new int[MAX_LEN][MAX_LEN];
    static boolean[][] isVisit = new boolean[MAX_LEN][MAX_LEN];

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
                                                                  
        n = Integer.parseInt(st.nextToken());
        q = Integer.parseInt(st.nextToken());
        len = 1 << n;

        for(int y = 0; y < len; y++){
            st = new StringTokenizer(br.readLine());
            for(int x = 0; x < len; x++){
                grid[y][x] = Integer.parseInt(st.nextToken());
            }
        }

        st = new StringTokenizer(br.readLine());

        for(int i = 0; i < q; i++){
            simulate(Integer.parseInt(st.nextToken()));
        }
        
        print();
    }

    static void simulate(int l){
        int partialGrid = 1 << l;
        // 2^L 크기의 부분 격자를 시계 방향으로 90도 회전시킨다.
        for(int y = 0; y < len; y+=partialGrid){
            for(int x = 0; x < len; x+=partialGrid){

                for(int i = 0; i < partialGrid / 2; i++){
                    int rotateSize = partialGrid - (2 * i);
                    rotate(y + i, x + i, rotateSize);
                }
            }
        }

        // 주변에 얼음이 2개 이하라면 얼음의 양을 1 줄인다.
        for(int y = 0; y < len; y++){
            for(int x = 0; x< len; x++){
                temp[y][x] = grid[y][x];
                if(grid[y][x]== 0) continue;
                int adjacentIceCount = 0;
                for(int d = 0; d < 4; d++){
                    int ny = y + dy[d];
                    int nx = x + dx[d];

                    if(inRange(ny, nx) && grid[ny][nx] != 0){
                        adjacentIceCount++;
                    }
                }

                if(adjacentIceCount < 3){
                    temp[y][x]--;
                }
            }
        }


        // 문제 2: 얼음이 녹는 조건의 동시성 누락
        // 방금 녹아서 0이 된 얼음이, 다음 칸(예: 바로 옆 칸)의 인접 얼음 개수를 셀 때 영향을 주게 됩니다
        for(int y = 0; y < len; y++){
            for(int x = 0; x< len; x++){
                grid[y][x] = temp[y][x];
            }
        }
    }

    static void rotate(int y, int x, int size){
        List<Integer> list = new ArrayList<>();
        for(int i = 0; i < size - 1; i++){
            list.add(grid[y][x + i]);
        } // 123

        for(int i = 0; i < size - 1; i++){
            grid[y][x + i] = grid[y + size - 1 - i][x];
        }
        // 9 17 25

        for(int i = 0; i < size - 1; i++){
            grid[y + 1 + i][x] = grid[y + size - 1][x + 1 + i];
        }
        // 26 27 28

        for(int i = 0; i < size - 1; i++){
            grid[y + size - 1][x + 1 + i] = grid[y + size - 2 - i][x + size - 1];
        }
        // 28 20 12

        for(int i = 0; i < size - 1; i++){
            grid[y + size - 2 - i][x + size - 1] = list.get(size - 2 - i);
        }
    }

    static boolean inRange(int y, int x){
        return y >= 0 && x >= 0 && y < len && x < len;
    }

    static void print(){
        int iceSum = 0;
        int maxPartialIce = 0;

        Queue<Pos> q = new ArrayDeque<>();

        for(int y = 0; y < len; y++){
            for(int x = 0; x < len; x++){
                if(isVisit[y][x]) continue;
                if(grid[y][x] == 0) continue;
                isVisit[y][x] = true;
                q.add(new Pos(y, x));
                iceSum += grid[y][x];
                int partialIce = 1;

                while(!q.isEmpty()){
                    Pos cur = q.poll();

                    for(int d = 0; d < 4; d++){
                        int ny = cur.y + dy[d];
                        int nx = cur.x + dx[d];

                        if(!inRange(ny, nx)) continue;
                        if(isVisit[ny][nx]) continue;
                        if(grid[ny][nx] == 0) continue; // 문제1: 이걸 안해서 maxParticalIce를 올림
                        isVisit[ny][nx] = true;
                        iceSum += grid[ny][nx];
                        partialIce++;
                        q.add(new Pos(ny, nx));
                    }
                }

                maxPartialIce = Math.max(maxPartialIce, partialIce);
            }
        }

        System.out.println(iceSum);
        System.out.println(maxPartialIce);
    }

}
```

## 최적화 풀이
```java
import java.io.*;
import java.util.*;

public class Main {

	public static int n, q;
	public static int[][] map;
	public static int[] dx = { -1, 1, 0, 0 };
	public static int[] dy = { 0, 0, -1, 1 };
	public static int land, totalIce;

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		StringTokenizer st;
		st = new StringTokenizer(br.readLine());

		n = Integer.parseInt(st.nextToken());
		q = Integer.parseInt(st.nextToken());

		n = (int) Math.pow(2, n);
		map = new int[n][n];
		for (int i = 0; i < n; i++) {
			st = new StringTokenizer(br.readLine());
			for (int j = 0; j < n; j++) {
				map[i][j] = Integer.parseInt(st.nextToken());
			}
		}

		st = new StringTokenizer(br.readLine());
		int[] L = new int[q];
		for (int i = 0; i < q; i++) {
			L[i] = Integer.parseInt(st.nextToken());
		}

		for (int i = 0; i < q; i++) {
			// 파이어스톰 시전
			map = divide(L[i]);// 회전
			map = melt();// 얼음 녹이기
		}
		land = totalIce = 0;

		biggest();
		System.out.println(totalIce);
		System.out.println(land);

	}

	public static int[][] divide(int L) {
		int[][] tmp = new int[n][n];
		L = (int) Math.pow(2, L);
		for (int i = 0; i < n; i += L) {
			for (int j = 0; j < n; j += L) {
				rotate(i, j, L, tmp);
			}
		}
		return tmp;

	}

	public static void rotate(int x, int y, int L, int[][] tmp) {
		for (int i = 0; i < L; i++) {
			for (int j = 0; j < L; j++) {
				tmp[x + i][y + j] = map[x + L - 1 - j][y + i];
			}
		}
	}

	public static int[][] melt() {
		int[][] tmp = new int[n][n];
		for (int i = 0; i < n; i++) {
			tmp[i] = Arrays.copyOf(map[i], n);
		}

		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				int cnt = 0;
				if (map[i][j] == 0)
					continue;
				for (int k = 0; k < 4; k++) {
					int nx = i + dx[k];
					int ny = j + dy[k];
					if (nx >= 0 && nx < n && ny >= 0 && ny < n) {
						if (map[nx][ny] > 0) {
							cnt++;
						}
					}
				}
				if (cnt < 3)
					tmp[i][j]--;
			}
		}
		return tmp;
	}

	public static void biggest() {
		Queue<int[]> q = new LinkedList<>();
		boolean[][] visit = new boolean[n][n];

		for (int i = 0; i < n; i++) {
			for (int j = 0; j < n; j++) {
				totalIce += map[i][j];
				if (map[i][j] > 0 && !visit[i][j]) {
					q.add(new int[] { i, j });
					visit[i][j] = true;
					int cnt = 1;

					while (!q.isEmpty()) {
						int[] t = q.poll();
						int tx = t[0];
						int ty = t[1];

						for (int k = 0; k < 4; k++) {
							int nx = tx + dx[k];
							int ny = ty + dy[k];
							if (nx >= 0 && nx < n && ny >= 0 && ny < n) {
								if (map[nx][ny] > 0 && !visit[nx][ny]) {
									visit[nx][ny] = true;
									q.add(new int[] { nx, ny });
									cnt++;
								}
							}
						}

					}
					land = Math.max(land, cnt);
				}
			}
		}
	}
}
```
