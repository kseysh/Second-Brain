
## 나의 풀이
```java
import java.io.*;
import java.util.*;

class Main {

    static final int MAX_N = 50;
    static final int MAX_M = 50;
    static final int EMPTY = 0;
    static final int CLOCK_WISE = 0;
    static final int REVERSE_CLOCK_WISE = 1;

    static int n, m, t;
    static int[][] disk = new int[MAX_N + 1][MAX_M + 1];
    static List<int[]> removePos = new ArrayList<>();
    static int[] rotateValue = new int[MAX_N + 1]; // 0 ~ m - 1

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
                                                                  
        n = Integer.parseInt(st.nextToken()); // 원판 반지름 & 원판의 수
        m = Integer.parseInt(st.nextToken()); // 원판에 적혀 있는 M개의 정수
        t = Integer.parseInt(st.nextToken()); // 회전할 수

        for(int i = 1; i <= n; i++){
            st = new StringTokenizer(br.readLine());
            for(int j = 1; j <= m; j++){
                disk[i][j] = Integer.parseInt(st.nextToken());
            }
        }

        while(t-->0) {
            st = new StringTokenizer(br.readLine());                                                    
            int x = Integer.parseInt(st.nextToken()); // x의 배수인 원판을
            int d = Integer.parseInt(st.nextToken()); // d 방향으로 
            int k = Integer.parseInt(st.nextToken()); // k칸 회전시킨다.

            // 원판 회전
            for(int i = x; i <= n; i += x){
                if(d == CLOCK_WISE){
                    rotateValue[i] = (rotateValue[i] - k + m) % m;
                }else{
                    rotateValue[i] = (rotateValue[i] + k) % m;
                }
            }

            // --- 수정된 인접 확인 로직 ---
            
            // 1. 원판 간(상하) 인접 확인
            for(int i = 2; i <= n; i++){
                for(int j = 1; j <= m; j++){
                    int curInsideDiskIdx = (j + rotateValue[i - 1] - 1) % m + 1;
                    int curDiskIdx = (j + rotateValue[i] - 1) % m + 1;
                    if(disk[i][curDiskIdx] == EMPTY) continue;

                    if(disk[i - 1][curInsideDiskIdx] == disk[i][curDiskIdx]){
                        removePos.add(new int[]{i - 1, curInsideDiskIdx});
                        removePos.add(new int[]{i, curDiskIdx});
                    }
                }
            }

            // 2. 원판 내(좌우) 인접 확인
            for(int i = 1; i <= n; i++){
                for(int j = 1; j <= m; j++){
                    int nextJ = (j % m) + 1; // j가 m일 때 1로 순환되도록 처리
                    if(disk[i][j] != EMPTY && disk[i][j] == disk[i][nextJ]){
                        removePos.add(new int[]{i, j});
                        removePos.add(new int[]{i, nextJ});
                    }
                }
            }

            // ---------------------------

            if(removePos.isEmpty()){
                int sum = 0;
                int count = 0;
                for(int i = 1; i <= n; i++){
                    for(int j = 1; j <= m; j++){
                        if(disk[i][j] == EMPTY) continue;
                        sum += disk[i][j];
                        count++;
                    }
                }

                if(count > 0){
                    for(int i = 1; i <= n; i++){
                        for(int j = 1; j <= m; j++){
                            if(disk[i][j] == EMPTY) continue;
                            if (sum > disk[i][j] * count) {
                                disk[i][j]++;
                            } else if (sum < disk[i][j] * count) {
                                disk[i][j]--;
                            }
                        }
                    }
                }
            }else{
                for(int[] pos: removePos){
                    disk[pos[0]][pos[1]] = EMPTY;
                }
                removePos.clear();
            }
        }

        int sum = 0;
        for(int i = 1; i <= n; i++){
            for(int j = 1; j <= m; j++){
                sum += disk[i][j];
            }
        }
        System.out.print(sum);
    }
}
```
### 틀린 이유
- 배수라고 했는데 `i *= x`를 써버려서 거듭 제곱으로 계산함
- 회전한 원판 주변에서만 제거하는 줄 알았는데, 회전은 따로고 인접한 전체를 제거하는 것이었음
- float 계산시에는 곱셈을 통해 int로 변환해서 비교하자.
