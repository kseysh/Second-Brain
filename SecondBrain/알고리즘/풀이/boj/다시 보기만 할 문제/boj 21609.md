## 나의 풀이
```java
import java.io.*;
import java.util.*;

class Main {

    static class Group {
        int size, rainbowCount, targetY, targetX;

        Group(int s, int r, int y, int x){
            this.size = s;
            this.rainbowCount = r;
            this.targetY = y;
            this.targetX = x;
        }
    }

    static class Pos {
        int y, x;

        Pos(int y, int x){
            this.y = y;
            this.x = x;
        }
    }

    static final int EMPTY_BLOCK = -10;
    static final int RAINBOW_BLOCK = 0;
    static final int BLACK_BLOCK = -1;
    static final int MAX_N = 20;
    static final int MAX_M = 5;
    static final int[] dy = {0, 1, 0, -1};
    static final int[] dx = {1, 0, -1, 0};
    static int[][] grid = new int[MAX_N][MAX_N];
    static int[][] temp = new int[MAX_N][MAX_N];
    static int n, m;
    static int answer = 0;

    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());

        for(int y = 0; y < n; y++){
            st = new StringTokenizer(br.readLine());
            for(int x = 0; x < n; x++){
                grid[y][x] = Integer.parseInt(st.nextToken());
            }
        }

        while(simulate()){}

        System.out.print(answer);
    }

    static boolean simulate(){
        List<Group> groups = new ArrayList<>();

        boolean[][] isVisit = new boolean[n][n];

        for(int y = 0; y < n; y++){
            for(int x = 0; x < n; x++){
                isVisit[y][x] = false;
            }
        }

        // 크기가 가장 큰 그룹 찾기
        for(int y = 0; y < n; y++){
            for(int x = 0; x < n; x++){
                Set<Pos> rainbowPos = new HashSet<>();
                if(isVisit[y][x]) continue;
                int targetColor = grid[y][x];
                if(
                    targetColor == RAINBOW_BLOCK || 
                    targetColor == EMPTY_BLOCK ||
                    targetColor == BLACK_BLOCK
                ) continue;
                isVisit[y][x] = true;

                int groupSize = 1;
                int rainbowCount = 0;

                Queue<Pos> q = new ArrayDeque<>();
                q.add(new Pos(y, x));
                while(!q.isEmpty()){
                    Pos cur = q.poll();
                    for(int d = 0; d < 4; d++){
                        int ny = cur.y + dy[d];
                        int nx = cur.x + dx[d];
                        if(!inRange(ny, nx)) continue;
                        if(isVisit[ny][nx]) continue;
                        int color = grid[ny][nx];
                        if(color == RAINBOW_BLOCK){
                            rainbowCount++;
                            isVisit[ny][nx] = true;
                            groupSize++;    
                            q.add(new Pos(ny, nx));
                            rainbowPos.add(new Pos(ny, nx));
                        }else if(color == targetColor){
                            isVisit[ny][nx] = true;
                            groupSize++;    
                            q.add(new Pos(ny, nx));
                        }
                    }
                }
                if(groupSize > 1) groups.add(new Group(groupSize, rainbowCount, y, x));
                for(Pos p : rainbowPos){
                    isVisit[p.y][p.x] = false;
                }
            }
        }

        if(groups.isEmpty()) return false; // 그룹이 존재하지 않으면 게임 끝
        Group removeTargetGroup = groups.get(0);
        for(int i = 1; i < groups.size(); i++){
            Group curGroup = groups.get(i);
            if(removeTargetGroup.size == curGroup.size){
                if(removeTargetGroup.rainbowCount == curGroup.rainbowCount){
                    if(removeTargetGroup.targetY == curGroup.targetY){
                        if(removeTargetGroup.targetX < curGroup.targetX){
                            removeTargetGroup = curGroup;
                        }
                    }else if(removeTargetGroup.targetY < curGroup.targetY){
                        removeTargetGroup = curGroup;
                    }
                }else if(removeTargetGroup.rainbowCount < curGroup.rainbowCount){
                    removeTargetGroup = curGroup;
                }
            }else if(removeTargetGroup.size < curGroup.size){
                removeTargetGroup = curGroup;
            }
        }

        // 찾은 그룹 제거 후 점수 향상
        Queue<Pos> q = new ArrayDeque<>();
        q.add(new Pos(removeTargetGroup.targetY, removeTargetGroup.targetX));
        int targetColor = grid[removeTargetGroup.targetY][removeTargetGroup.targetX];
        grid[removeTargetGroup.targetY][removeTargetGroup.targetX] = EMPTY_BLOCK;
        while(!q.isEmpty()){
            Pos cur = q.poll();
            for(int d = 0; d < 4; d++){
                int ny = cur.y + dy[d];
                int nx = cur.x + dx[d];
                if(!inRange(ny, nx)) continue;
                int color = grid[ny][nx];
                if(color == RAINBOW_BLOCK || color == targetColor){
                    grid[ny][nx] = EMPTY_BLOCK;
                    q.add(new Pos(ny, nx));    
                }
            }
        }
        answer += (int)Math.pow(removeTargetGroup.size, 2);

        gravity();
        rotate();
        gravity();
        return true;
    }

    static boolean inRange(int y, int x){
        return y >= 0 && x >= 0 && y < n && x < n;
    }

    static void gravity() {
        if(n == 1) return;

        for(int x = 0; x < n; x++){
            for(int y = n - 2; y >= 0; y--){
                int color = grid[y][x];
                // Empty Block과 Black Block은 gravity 제외
                if(color == EMPTY_BLOCK || color == BLACK_BLOCK) continue;

                int targetY = y;
                for(int ny = y + 1; ny < n; ny++){
                    if(grid[ny][x] == EMPTY_BLOCK) targetY = ny;
                    else break;
                }
                
                grid[y][x] = EMPTY_BLOCK;
                grid[targetY][x] = color;
            }
        }
    }

    static void rotate(){
        if(n == 1) return;

        for(int y = 0; y < n; y++){
            for(int x = 0; x < n; x++){
                temp[n - 1 - x][y] = grid[y][x];
            }
        }

        for(int y = 0; y < n; y++){
            for(int x = 0; x < n; x++){
                grid[y][x] = temp[y][x];
            }
        }
    }

    static void debug(){
        System.out.println("--------------");
        for(int y = 0; y < n; y++){
            for(int x = 0; x < n; x++){
                Integer value = grid[y][x];
                String s = value.toString();
                if(value == EMPTY_BLOCK) s = "x";
                System.out.print(s);
                System.out.print(' ');
            }
            System.out.println();
        }
        System.out.println("--------------");
    }
}
```
