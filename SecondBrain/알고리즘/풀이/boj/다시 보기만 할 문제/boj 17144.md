static void doClean(){
        // 반시계 회전
        for (int y = rccy - 2; y >= 0; y--) {
            map[y + 1][0] = map[y][0];
        }

        for (int x = 1; x < c; x++) {
            map[0][x - 1] = map[0][x];
        }

        for (int y = 1; y <= rccy; y++) {
            map[y - 1][c - 1] = map[y][c - 1];
        }

        for (int x = c - 1; x >= 0; x--) {
            map[rccy][x + 1] = map[rccy][x];
        }

        map[rccy][1] = 0;

        // 시계 회전
        for (int y = ccy + 2; y < r; y++) {
            map[y - 1][0] = map[y][0];
        }

        for (int x = 1; x < c; x++) {
            map[r - 1][x - 1] = map[r - 1][x];
        }

        for (int y = r; y >= ccy; y--) {
            map[y][c - 1] = map[y - 1][c - 1];
        }

        for (int x = c - 1; x >= 0; x--) {
            map[ccy][x + 1] = map[ccy][x];
        }

        map[ccy][1] = 0;
    }
