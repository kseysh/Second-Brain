#include <string>
#include <vector>

using namespace std;

int calculateCompressedSize(int n, const string &s){
    int ssize = s.size();
    string result = "";
    int count = 1;
    for(int i = 0; i < ssize;){
        if(i + n >= ssize){
            result += s.substr(i);
            return result.size();
        }
        string compare = s.substr(i, n);
        int count = 1;
        while(true){
            if(i + count * n >= ssize){
                break;
            }
            if(s.substr(i + count * n, n) == compare){
                count++;
            }else{
                break;
            }
        }
        
        if(count != 1){
            result += to_string(count) + compare;
        }else{
            result += s.substr(i, n);
        }
        i += n * count;
    }
    return result.size();
    
}

int solution(string s) {
    int ssize = s.size();
    int answer = ssize;
    for(int unit = 1; unit <= ssize / 2; unit++){
        answer = min(answer, calculateCompressedSize(unit, s));
    }
    
    return answer;
}
