# 나의 풀이
```cpp
#include <string>
#include <vector>

using namespace std;

int calculateCompressedSize(int n, const string &s){
    int ssize = s.size();
    string result = "";
    int count = 1;
    for(int i = 0; i < ssize;){
        if(i + n >= ssize){
            result += s.substr(i);
            return result.size();
        }
        string compare = s.substr(i, n);
        int count = 1;
        while(true){
            if(i + count * n >= ssize){
                break;
            }
            if(s.substr(i + count * n, n) == compare){
                count++;
            }else{
                break;
            }
        }
        
        if(count != 1){
            result += to_string(count) + compare;
        }else{
            result += s.substr(i, n);
        }
        i += n * count;
    }
    return result.size();
    
}

int solution(string s) {
    int ssize = s.size();
    int answer = ssize;
    for(int unit = 1; unit <= ssize / 2; unit++){
        answer = min(answer, calculateCompressedSize(unit, s));
    }
    
    return answer;
}
```
풀이법은 괜찮다고 느끼는데, 시간이 너무 오래걸렸다.
# 다른 사람의 풀이
```cpp
#include <string>
#include <vector>
#include <iostream>

using namespace std;

vector<string> convert(string s, int n)
{
    vector<string> v;
    for(int i = 0 ; i <s.length() ; i+=n)
    {
        v.push_back(s.substr(i,n));
    }
    return v;
}

int solution(string s) {
    int answer = 0;
    vector<string> tok;
    string before;
    int cnt = 1;
    int min = s.length();
    string str="";
    for(int j = 1 ; j <= s.length()/2 ; j++)
    {
        tok = convert(s,j);
        str = "";
        before = tok[0];
        cnt = 1;
        for(int i = 1 ; i < tok.size() ; i++)
        {
            if(tok[i] == before) cnt++;
            else
            {
                if(cnt != 1) str += to_string(cnt);
                str += before;
                before = tok[i];
                cnt = 1;
            }
        }
        if(cnt != 1)str += to_string(cnt);
        str += before;  
        min = min<str.length() ? min : str.length();
    }
    cout<<str;

    return min;
}
```
