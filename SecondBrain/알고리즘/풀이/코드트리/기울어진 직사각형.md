```cpp
#include <iostream>
#include <algorithm>
#define MAX_N 20

using namespace std;

int n;
int dx[4] = {-1, -1, 1, 1};
int dy[4] = {1, -1, -1, 1};
int map[MAX_N][MAX_N];

bool inRange(int x, int y){
    return 0 <= x && x < n && 0 <= y && y < n;
}

int getScore(int y, int x, int w, int h){
    int moveNum[4] = {w, h, w, h};
    int sum = 0;

    for(int d = 0; d < 4; d++){
        for(int q = 0; q < moveNum[d]; q++){
            x += dx[d];
            y += dy[d];
            if(!inRange(x, y)) return 0;

            sum += map[x][y];
        }
    }

    return sum;
}


int main() {
    cin >> n;

    for(int y = 0; y < n; y++){
        for(int x = 0; x < n; x++){
            cin >> map[y][x];
        }
    }

    int result = 0;

    for(int y = 0; y < n; y++){
        for(int x = 0; x < n; x++){
            for(int w = 1; w < n; w++){
                for(int h = 1; h < n; h++){
                    result = max(result, getScore(y, x, w, h));
                }
            }
        }
    }

    cout << result;
    return 0;
}
```

dx dy 테크닉에 대해서 평소와 다른 방식으로 접근한 방식이다. 
여기서 moveNum 배열을 사용한 테크닉을 다시 더 확인해보자
