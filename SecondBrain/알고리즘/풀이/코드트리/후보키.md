[[비트마스킹]]

## 다른 사람의 풀이

```cpp
#include <set>
#include <string>
#include <vector>

 using namespace std;
bool canFk(vector<int> &vec, int now){ // 최소성을 체크하기 위해 answer 중에서 & 연산을 했을 때 answer와 값이 같으면 안됨
    for(int i = 0; i < vec.size(); i++){
        if((vec[i] & now) == vec[i]) return false;
    }
    return true;
}
int solution(vector<vector<string>> relation) {
    int r = relation.size();
    int c = relation[0].size();
    vector<int> ans;
    for(int i = 1; i < (1 << c); i++){ // 모든 부분집합에 대해서 확인하기 위함
        set<string> s;
        for(int j = 0; j < r; j++){ // 모든 row에 대해 중복이 존재하는지 확인한다.
            string now = "";
            for(int k = 0; k < c; k++){ // 부분집합에 대해 set에 넣어 확인한다.
                if(i & (1 << k)) now += relation[j][k] + '&';
            }
            s.insert(now);
        }
        if(s.size()==r && canFk(ans,i)) ans.push_back(i); // set의 size가 r과 같으면 중복이 없다는 뜻 && 최소성 만족하는지 체크
    }
    return ans.size();
}
```
