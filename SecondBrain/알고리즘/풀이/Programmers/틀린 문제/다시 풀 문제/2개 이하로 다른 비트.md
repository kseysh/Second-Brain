# 나의 풀이
```cpp
#include <string>
#include <vector>

using namespace std;

long long powerTwo[51];

int getN(long long num){
    for(int i = 0; i <= 50; i++){
        if(powerTwo[i] == num + 1) return i;
    }
    return -1;
}

void init(){
    powerTwo[0] = 1;
    for(int i = 1; i <= 50; i++){
        powerTwo[i] = powerTwo[i-1] * 2;
    }
}

long long convertPowerTwo(int n){
    long long cur = 1;
    for(int i = 0; i < n; i++){
        cur *= 2;
    }
    return cur;
}

vector<long long> solution(vector<long long> numbers) {
    
    init();
    vector<long long> answer;
    for(long long num : numbers){
        int n = getN(num);
        if(n != -1){ // 11111로 끝남
            answer.push_back(num + convertPowerTwo(n - 1));
        }else{ // 
            long long cur = num;
            for(int i = 1; i <= 50; i++){
                if(cur % convertPowerTwo(i) == 0){ // 2로 나눠봤을 때 나눠지면
                    if(i == 1) answer.push_back(num + convertPowerTwo(i - 1)); // 첫 번째 비트가 0이라는 것
                    else answer.push_back(num + convertPowerTwo(i - 1) - convertPowerTwo(i - 2)); // 첫 번째 비트가 0이라는 것
                    
                    break;
                }
                cur -= convertPowerTwo(i - 1);
            }
        }
    }
    
    return answer;
}
```

#include <vector>
vector<long long> solution(vector<long long> numbers) {
    vector<long long> answer;
    for (long long number : numbers) {
        long long bit = 1;
        while ((number & bit) > 0) bit <<= 1;
        answer.push_back(number + bit - (bit >> 1));
    }
    return answer;
}
