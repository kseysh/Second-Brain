## 다른 사람의 풀이
```cpp
#include <string>
#include <vector>
#include <cstring>
using namespace std;

long long y = 0;
bool check[64]; // 숫자의 i번째 비트가 1인지 저장 
bool visit[64]; // DFS 방문 여부

void dfs(int now, int dx){
    visit[now] = true; 
    y += (1LL << now); // 방문했다면 그만큼 값을 더해줌
    for (int next : {now - dx, now + dx}){
	    if (check[next] && !visit[next]){
		    dfs(next, dx / 2); // 다음 dx는 현재 dx의 절반임
	    }
    }
}

vector<int> solution(vector<long long> numbers) {
    vector<int> answer;
    for (long long x : numbers){
        bool flag = false;
        memset(check, 0, sizeof(check));
        // 숫자를 비트 배열로 변환한다.
        for (int k = 0; k < 63; k++){ // 포화 이진 트리는 63개의 bit가 존재, 따라서 62번 밀어야 함
	        if ((1LL << k) & x){ // long long value까지 갈 수 있음에 유의
		        check[k] = true;
	        }
        }

        for (int root : {0, 1, 3, 7, 15, 31}){ // 포화 이진트리에서는 루트 노드가 특정 위치에 있어야 함 (2^k - 1 형식)
	        if (check[root]){
	            y = 0;
	            memset(visit, 0, sizeof(visit));
	            dfs(root, (root + 1) / 2); // dx는 (root + 1) / 2임 (0과 자신의 절반이 왼쪽 자식일 것이므로)
	            if(x == y) flag = true; // x == y면 root가 진짜 루트라는 것 && dfs로 모든 노드를 내려갈 수 있다는 것
	            if (flag) break;
	        }
        }
        answer.push_back(flag);
    }
    return answer;
}
```
## 알아둘 것
이진트리 dfs로 방문하는 방법
1LL을 이용해 overflow 발생하지 않도록 유의하는 것
십진수를 이진수 배열로 변환하는 방법
