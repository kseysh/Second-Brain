## 다른 사람의 풀이
```cpp
#include <string>
#include <vector>
#include <cstring>
using namespace std;

long long y = 0;
bool check[64]; // 숫자의 i번째 비트가 1인지 저장 
bool visit[64]; // DFS 방문 여부

void dfs(int now, int dx){
    visit[now] = true; y += (1LL << now);
    for (int next : {now - dx, now + dx}){
	    if (check[next] && !visit[next]){
		    dfs(next, dx / 2);
	    }
    }
}

vector<int> solution(vector<long long> numbers) {
    vector<int> answer;
    for (long long x : numbers){
        bool flag = false;
        memset(check, 0, sizeof(check));
        for (int k = 0; k < 63; k++){
	        if ((1LL << k) & x){
		        check[k] = true;
	        }
        }

        for (int root : {0, 1, 3, 7, 15, 31}){
	        if (check[root]){
	            y = 0;
	            memset(visit, 0, sizeof(visit));
	            dfs(root, (root + 1) / 2);
	            if (flag = (x == y)) break;
	        }
        }
        answer.push_back(flag);
    }
    return answer;
}
```
