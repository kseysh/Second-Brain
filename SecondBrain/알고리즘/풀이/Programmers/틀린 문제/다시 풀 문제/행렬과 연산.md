## 다른 사람의 풀이
```cpp
#include <string>
#include <vector>
#include <list>
using namespace std;

vector<vector<int>> solution(vector<vector<int>> rc, vector<string> operations) {
    vector<vector<int>> answer;
    list<int> left, right;
    list<int> mid[rc.size()];
    for(auto rc_sub: rc){
        left.push_back(rc_sub[0]);
        right.push_back(rc_sub[rc_sub.size()-1]);
    }
    int i =0;
    for(auto rc_sub: rc){
        list<int> dest(rc_sub.begin()+1, rc_sub.end()-1);
        mid[i++] = dest;
    }
    int baseT = 0;
    int baseB = rc.size()-1;
    for(auto operation : operations) {
        if (operation == "ShiftRow") {
            baseT = baseT-1 <= rc.size() ? baseT-1 : rc.size() - 1;
            baseB = baseB-1 <= rc.size() ? baseT-1 : rc.size() - 1;
            left.push_front(left.back());
            left.pop_back();
            right.push_front(right.back());
            right.pop_back();
        }
        if (operation == "Rotate") {
            mid[baseT].push_front(left.front());
            left.pop_front();

            right.push_front(mid[baseT].back());
            mid[baseT].pop_back();

            mid[baseB].push_back(right.back());
            right.pop_back();

            left.push_back(mid[baseB].front());
            mid[baseB].pop_front();
        }
    }

    for(int i=0;i<rc.size();i++){
        int base = baseT+i >= rc.size() ? baseT+i - rc.size() : baseT+i;

        rc[i][0] = left.front();
        left.pop_front();
        rc[i][rc[0].size()-1] = right.front();
        right.pop_front();
        std::copy(mid[base].begin(), mid[base].end(), rc[i].begin()+1);
    }



    return rc;
}
```
