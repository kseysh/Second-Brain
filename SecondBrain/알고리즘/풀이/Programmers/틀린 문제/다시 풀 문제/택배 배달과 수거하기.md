## 다른 사람의 풀이
```cpp
#include <string>
#include <vector>

using namespace std;

long long solution(int cap, int n, vector<int> deliveries, vector<int> pickups) {
    long long answer = 0;

    int del = 0;
    int pic = 0;
    for (int i = n - 1; i >= 0; i--) { // 가장 먼 집부터 탐색
        if (deliveries[i] != 0 || pickups[i] != 0) { // 배달할 것이나 수거할 것이 있으면 들른다.
            int cnt = 0;
            while (del < deliveries[i] || pic < pickups[i]) {
                ++cnt;
                del += cap;
                pic += cap;
            }
            del -= deliveries[i];
            pic -= pickups[i];
            answer += (1LL * cnt) * (1LL * (i + 1)) * 2; // 계산 시 다운캐스팅 되는 것 유의
        }
    }

    return answer;
}
```
