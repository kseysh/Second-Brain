# 나의 풀이
```cpp
#include <string>
#include <vector>

using namespace std;

vector<int> solution(vector<int> numbers) {
    int nSize = numbers.size();
    vector<int> answer(nSize);
    int cur = 0;
    int prev = 0;
    int prevResultIdx = -1;
    for(int i = nSize - 1; i >= 0; i--){
        cur = numbers[i];
        if(prev <= cur){ // 여기서 핑퐁해야함
            if(prevResultIdx == -1){
                answer[i] = prevResultIdx;
                prev = cur;
                continue;
            } 
            
            // 계속 뒷수를 보면서 자신보다 큰 수를 찾음
            while(true){
                if(prevResultIdx == -1){
                    answer[i] = -1;
                    break;
                }
                if(numbers[prevResultIdx] > cur){
                    answer[i] = prevResultIdx;
                    break;
                }
                prevResultIdx = answer[prevResultIdx];
            }
        }else{ // cur이 작으면 prev가 뒷수
            prevResultIdx = i + 1;
            answer[i] = prevResultIdx;
        }
        prev = cur;
    }
    
    
    vector<int> result(nSize);
    for(int i = 0; i < nSize; i++){
        if(answer[i] == -1){
            result[i] = -1;
        }else{
            result[i] = numbers[answer[i]];
        }
    }
    return result;
}
```
# 다른 사람의 풀이
```cpp
#include <string>
#include <vector>
#include <stack>

using namespace std;

vector<int> solution(vector<int> numbers) {
    vector<int> answer(numbers.size(), -1);
    stack<int> stk;

    for(int i=0; i<numbers.size(); i++) { 
        while(!stk.empty() && numbers[stk.top()] < numbers[i]) {
            answer[stk.top()] = numbers[i];
            stk.pop();
        }
        stk.push(i);
    }

    return answer;
}
```
