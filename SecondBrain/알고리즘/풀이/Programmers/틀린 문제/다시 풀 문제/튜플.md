# 내 풀이
```cpp
#include <string>
#include <vector>

using namespace std;

bool isUsed[100'001];
vector<string> v(501);
vector<int> answer;
int maxElementCount = 1;

void fillV(string &s){
    int sSize = s.size();
    
    int startSetIdx = 1;
    int commaCount = 0;
    for(int i = 1; i < sSize-1; i++){ // 맨 앞 맨 뒤 중괄호는 뺀다.
        if(s[i] == '{'){
            startSetIdx = i;
            continue;
        }
        if(s[i] == '}'){
            // number vector에 넣는다.
            v[commaCount] = s.substr(startSetIdx + 1, i - startSetIdx - 1);
            i++; // 집합 간 콤마를 건너뛰기 위함
            commaCount = 0;
            continue;
        }
        if(s[i] == ','){
            commaCount++;
            maxElementCount = max(maxElementCount , commaCount + 1);
        }  
    }
}

int getNotDuplicated(string &s){
    int commaIdx = 1;
    int sSize = s.size();
    int startNumIdx = 0;
    for(int i = 0; i < sSize; i++){
        if(s[i] == ','){
            int num = stoi(s.substr(startNumIdx, i - startNumIdx));
            if(!isUsed[num]){
                isUsed[num] = true;
                return num;
            } 
            startNumIdx = i+1;
        }
    }
    int num = stoi(s.substr(startNumIdx, s.size() - startNumIdx));
    isUsed[num] = true;
    return num;
}

void findResult(){
    for(int i = 0; i < maxElementCount; i++){
        answer.push_back(getNotDuplicated(v[i]));
    }
}

vector<int> solution(string s) {
    fillV(s);    
    findResult();
    return answer;
}
```
