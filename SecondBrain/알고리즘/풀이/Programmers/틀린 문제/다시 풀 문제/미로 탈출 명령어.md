## 다른 사람의 풀이
```cpp
#include <string>
#include <vector>
using namespace std;


int distance(int y, int x, int r, int c){ // 맨해튼 거리 계산
    return abs(y-r)+abs(x-c);
}


string solution(int n, int m, int x, int y, int c, int r, int k) {
    string answer = "";
    while(k > 0){
        --k;
        if(x < n && distance(y,x+1,r,c) <= k) {answer += "d"; x+=1;}
        else if(y > 1 && distance(y-1,x,r,c) <= k) {answer += "l"; y-=1;}
        else if(y < m && distance(y+1,x,r,c) <= k) {answer += "r"; y+=1;}
        else if(x > 1 && distance(y,x-1,r,c) <= k) {answer += "u"; x-=1;}
        else if(k == 0) return "impossible";
    }
    return answer;
}
```
