#include <string>
#include <vector>
#include <stack>

using namespace std;

stack<int> stk;

vector<int> solution(vector<int> prices) {
    vector<int> answer(prices.size());
    
    for(int i = 0; i < prices.size(); i++){
        int price = prices[i];
        int count = 1;
        while(!stk.empty() && prices[stk.top()] > price){
            answer[stk.top()] = i - stk.top(); 
            stk.pop();
        }
        stk.push(i);
    }
    
    while(!stk.empty()){
        answer[stk.top()] = prices.size() - stk.top() - 1;
        stk.pop();
    }
    
    return answer;
}
