# 나의 풀이
```cpp
#include <string>
#include <list>
#include <vector>

using namespace std;

list<char> lt;
list<char>::iterator curIter, prevIter;
string solution(string number, int k) {
    
    for(char c : number){
        lt.push_back(c);
    }
    
    curIter = lt.begin();
    for(; curIter != lt.end();){
        if(curIter == lt.begin()){
            curIter++;
            continue;
        }
        prevIter = prev(curIter);
        if(*prevIter < *curIter){
            curIter = lt.erase(prevIter);
            if(--k == 0) break;
        }else{
            curIter++;   
        }
        
    }
    
    string answer = "";
    for(char c : lt){
        answer.push_back(c);
    }
    
    return answer.substr(0, answer.size() - k);
}
```

# 다른 사람의 풀이
```cpp
#include <string>
#include <vector>
using namespace std;

string solution(string number, int k) {
    string answer = "";

    char biggest = 0;
    int idx = 0;

    for (int i = 0; i < number.length(); ++i) {
        if (biggest < number[i]) {
            biggest = number[i];
            idx = i;
        }

        if (i == k) {
            i = idx;
            ++k;
            answer.push_back(biggest);
            biggest = 0;
        }
    }

    return answer;
}
```

```cpp
#include <string>
using namespace std;

string solution(string number, int k) {
    string answer = "";
    int toPick = number.size() - k; // 뽑아야 할 자릿수
    int start = 0;                  // 탐색 시작 위치

    while (toPick > 0) {
        char biggest = 0;
        int idx = start;

        // [start, number.size() - toPick] 구간에서 가장 큰 수 탐색
        for (int i = start; i <= number.size() - toPick; i++) {
            if (biggest < number[i]) {
                biggest = number[i];
                idx = i;
            }
        }

        // 찾은 가장 큰 수 추가
        answer.push_back(biggest);

        // 다음 탐색 시작 위치 갱신
        start = idx + 1;
        toPick--;
    }

    return answer;
}
```
