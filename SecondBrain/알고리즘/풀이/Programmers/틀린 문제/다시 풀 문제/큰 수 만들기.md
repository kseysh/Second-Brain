# 나의 풀이
```cpp
#include <string>
#include <list>
#include <vector>

using namespace std;

list<char> lt;
list<char>::iterator curIter, prevIter;
string solution(string number, int k) {
    
    for(char c : number){
        lt.push_back(c);
    }
    
    curIter = lt.begin();
    for(; curIter != lt.end();){
        if(curIter == lt.begin()){
            curIter++;
            continue;
        }
        prevIter = prev(curIter);
        if(*prevIter < *curIter){
            curIter = lt.erase(prevIter);
            if(--k == 0) break;
        }else{
            curIter++;   
        }
        
    }
    
    string answer = "";
    for(char c : lt){
        answer.push_back(c);
    }
    
    return answer.substr(0, answer.size() - k);
}
```

# 다른 사람의 풀이
```cpp
#include <string>
#include <vector>
using namespace std;

string solution(string number, int k) {
    string answer = "";

    char biggest = 0;
    int idx = 0;

    for (int i = 0; i < number.length(); ++i) {
        if (biggest < number[i]) {
            biggest = number[i];
            idx = i;
        }

        if (i == k) {
            i = idx;
            ++k;
            answer.push_back(biggest);
            biggest = 0;
        }
    }

    return answer;
}
```
