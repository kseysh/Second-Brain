## 다시 푼 풀이
```java
import java.util.*;

class Solution {
    
    class Price{
        int time, amount;
        
        Price(int time, int amount){
            this.time = time;
            this.amount = amount;
        }
    }
    
    public int[] solution(int[] prices) {        
        Stack<Price> stk = new Stack<>();
        int[] answer = new int[prices.length];
        
        for(int i = 0; i < prices.length; i++){                         
            while(!stk.isEmpty()){
                if(stk.peek().amount > prices[i]){ // pop
                    Price price = stk.pop();
                    answer[price.time] = i - price.time;
                }else{
                    break;
                }    
            }
            stk.add(new Price(i, prices[i]));
        }
        
        while(!stk.isEmpty()){
            Price price = stk.pop();
            answer[price.time] = prices.length - 1 - price.time;
        }
        
        return answer;
    }
}
```
