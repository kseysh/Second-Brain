class Solution {
    long maxLeaves = 1;

    public long solution(int dist_limit, int split_limit) {
        dfs(1, 1, 0, dist_limit, split_limit);
        return maxLeaves;
    }

    // 현재 분배도, 현재 층의 노드 수, 지금까지 사용한 분배 노드 수
    void dfs(long p, long n, long used, int dist_limit, int split_limit) {
        // 최대 리프 노드 개수 갱신
        if (n > maxLeaves) {
            maxLeaves = n;
        }

        // 현재 층에서 분배 노드로 만들 노드의 수(k) 결정
        // 남은 분배 노드 허용치와 현재 층의 노드 수 중 최대한 많이 분배
        long k = Math.min(n, dist_limit - used);

        // 더 이상 분배 노드를 추가할 수 없다면 이 가지(Branch)는 탐색 종료
        if (k == 0) return;

        for(int i = 2; i <= 3; i++){
            if (p * i <= split_limit) {
                dfs(p * i, n + k * (i - 1), used + k, dist_limit, split_limit);
            }
        }
        
    }
}
