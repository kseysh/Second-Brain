## 나의 풀이
```cpp
#include <string>
#include <vector>
using namespace std;

vector<int> maxInfo; // 각 점수는 maxInfo의 값을 넘지 않음
vector<int> selectedScores(11, 0);
vector<int> answer(11, 0);
vector<int> rivalScores;
vector<bool> isSelected(11, false);
int n;
int maxDiff = 0;

void switchAnswer(const vector<int> &scores){
    for(int i = 0; i <= 10; i++){
        answer[i] = scores[i];
    }
}

void checkDiff(int count){
    int rivalSum = 0;
    int mySum = 0;
    
    for(int i = 0; i <= 10; i++){
        if(isSelected[i]){
            selectedScores[i] = maxInfo[i];
        }else{
            selectedScores[i] = 0;
        }
    }
    
    selectedScores[10] += (n - count);
    
    for(int i = 0; i <= 10; i++){
        int rivalScore = rivalScores[i];
        int myScore = selectedScores[i];
        if(rivalScore == 0 && myScore == 0) continue;
            
        if(rivalScore >= myScore){
            rivalSum += (10 - i);
        }else {
            mySum += (10 - i);
        }
    }
    
    if(maxDiff < mySum - rivalSum){
        maxDiff = mySum - rivalSum;
        switchAnswer(selectedScores);
    }else if(maxDiff == mySum - rivalSum){
        for(int i = 10; i >= 0; i--){
            if(answer[i] < selectedScores[i]){
                switchAnswer(selectedScores);
                break;
            }else if(answer[i] > selectedScores[i]){ // 변경하지 않음
                break;
            }
        }
    }
}

void simulate(int count){
    for(int i = 0; i <= 10; i++){
        if(count + maxInfo[i] <= n && !isSelected[i]){
            isSelected[i] = true;
            simulate(count + maxInfo[i]);
            isSelected[i] = false;
        }else{
            checkDiff(count);
        }
    }
}

vector<int> solution(int N, vector<int> info) {
    rivalScores = info;
    n = N;
    for(int i : info){
        maxInfo.push_back(i + 1);
    }
    
    simulate(0);
    
    if(maxDiff == 0) return {-1};
    return answer;
}
```
내 풀이에 대한 확신이 부족했음 => 완탐이라 생각하고 제대로 더 고민했어야 함
첫 풀이에서 몇 개 시간 초과났음 => 조금 느리다고 생각했다면 다른 상황도 테스트 했어야 함
