

# 확인한 풀이
```cpp
#include <string>
#include <vector>
#include <deque>
#define INF 1e9

using namespace std;

vector<int> solution(int m, int n, int h, int w, vector<vector<int>> drops) {
    
    vector<vector<int>> map(m, vector<int>(n, INF));
    vector<vector<int>> windowMin(m, vector<int>(n, INF));
    
    for(int time = 1; time <= drops.size(); time++){
        
        int dropY = drops[time - 1][0];
        int dropX = drops[time - 1][1];
        
        if(map[dropY][dropX] > time){
            map[dropY][dropX] = time;
        }
    }
        
    // x축 최솟값 찾기
    for(int y = 0; y < m; y++){
        deque<int> dq;
        for(int x = 0; x < n; x++){
            // 윈도우를 벗어나면 pop
            if(!dq.empty() && dq.front() < x - w + 1) dq.pop_front(); 
            
            // 최솟값 유지
            while(!dq.empty() && map[y][dq.back()] >= map[y][x]) dq.pop_back();
            
            dq.push_back(x); // deque에는 인덱스 저장
            
            // 틀린부분1: 실제 최솟값을 저장한다.
            if (x >= w - 1) { 
                windowMin[y][x - w + 1] = map[y][dq.front()]; 
            }
        }
    }
    
    // y축 누적합 & 결과 찾기
    int maxValue = -1;
    vector<int> answer(2, 0);
    for(int x = 0; x < n - w + 1; x++){
        deque<int> dq;
        for(int y = 0; y < m; y++){
            // 윈도우를 벗어나면 pop
            if(!dq.empty() && dq.front() < y - h + 1) dq.pop_front(); 
            
            // 최솟값 유지
            while(!dq.empty() && windowMin[dq.back()][x] >= windowMin[y][x]) dq.pop_back();
            
            dq.push_back(y);
            
            if(y >= h - 1){
                int curMax = windowMin[dq.front()][x];
                int r = y - h + 1;
                int c = x;
                if(curMax > maxValue){
                    maxValue = curMax;
                    answer[0] = r;
                    answer[1] = c;
                } else if(curMax == maxValue) {
                    if (r < answer[0]) {
                        answer[0] = r;
                        answer[1] = c;
                    } else if (r == answer[0] && c < answer[1]) {
                        answer[0] = r;
                        answer[1] = c;
                    }
                }   
            }
        }
    }
        
    return answer;
}
```

이차원 슬라이딩 윈도우 문제. 
1. w길이 만큼만 슬라이딩 윈도우 진행 
	1. deque를 활용하여 row에서의 윈도우 크기 만큼의 인덱스를 저장한다. 
	2. 인덱스를 저장하면서 deque의 front가 최솟값이도록 구성한다.
2. h길이에서 가장 작은 값 탐색
