```java
import java.util.*;

class Solution {
    public int solution(int n, int k, int[] enemy) {
        int answer = 0;
        PriorityQueue<Integer> pq = new PriorityQueue<>(Collections.reverseOrder());
    
        int sum = 0;
        int enemyCount = enemy.length;
        for(int i = 0; i < enemyCount; i++){
            pq.add(enemy[i]);
            sum += enemy[i];
            if(sum > n){
                if(k > 0){
                    sum -= pq.poll(); // 무적권 사용
                    k--;
                }else{
                    return answer;
                }
            }
            answer++;
        }
        
        
        return answer;
    }
}
```
