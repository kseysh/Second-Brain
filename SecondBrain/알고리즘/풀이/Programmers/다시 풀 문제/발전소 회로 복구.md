```java
import java.util.*;

class Solution {
    
    final int INF = 40 * 40 * 40;
    final int CANT_REACH = Integer.MAX_VALUE;
    final char PATH = '.';
    final char WALL = '#';
    final char ELEVATOR = '@';
    final int[] dy = {0, 1, 0, -1};
    final int[] dx = {1, 0, -1, 0};
    
    int n, m, k, elevatorY, elevatorX, answer;
    int[][] dist;
    int[][] adjacencyMatrix;
    // int[] indegrees;
    int[] requiredMask; // i번 패널을 위해 필요한 선행 패널 비트마스크
    // List<Integer>[] isNeeded; // j가 i 열기 위해 필요한지
    int[][] dp;
    
    public int solution(int h, String[] grid, int[][] panels, int[][] seqs) {
        this.n = grid.length;
        this.m = grid[0].length();
        
        findElevator(grid);
        dist = caculateDist(grid, elevatorY, elevatorX);
        makeAdjacencyMatrix(grid, panels);
        // makeIndegrees(seqs);
        makeRequiredMasks(seqs);
        
        dp = new int[k][1 << k];
        for(int i = 0; i < k; i++) Arrays.fill(dp[i], -1);
        
        int answer = Integer.MAX_VALUE;
        for(int i = 0; i < k; i++){
            if(requiredMask[i] == 0){
                answer = Math.min(answer, adjacencyMatrix[0][i] + tsp(i, 1 << i));
            }
        }
        return answer;
    }
    
    void findElevator(String[] grid){
        for(int i = 0; i < n; i++){
            for(int j = 0; j < m; j++){
                if(grid[i].charAt(j) == ELEVATOR){
                    elevatorY = i;
                    elevatorX = j;
                    return;
                }
            }
        }
        
        return;
    }
    
    int[][] caculateDist(String[] grid, int sy, int sx){
        int[][] dist = new int[n][m];
        
        for(int i = 0; i < n; i++){
            Arrays.fill(dist[i], CANT_REACH);
        }
        
        Queue<int[]> q = new ArrayDeque<>();
        q.add(new int[]{sy, sx});
        dist[sy][sx] = 0;
        while(!q.isEmpty()){
            int[] curPos = q.poll();
            int cy = curPos[0];
            int cx = curPos[1];
            
            for(int d = 0; d < 4; d++){
                int ny = cy + dy[d];
                int nx = cx + dx[d];
                if(
                    inRange(ny, nx) && 
                    grid[ny].charAt(nx) != WALL && 
                    dist[cy][cx] + 1 < dist[ny][nx]
                ){
                    dist[ny][nx] = dist[cy][cx] + 1;
                    q.add(new int[]{ny, nx});    
                }
            }
        }
        
        return dist;
    }
    
    void makeAdjacencyMatrix(String[] grid, int[][] panels){
        k = panels.length;
        adjacencyMatrix = new int[k][k];
        
        for(int i = 0; i < k; i++){
            int sf = panels[i][0];
            int sy = panels[i][1] - 1;
            int sx = panels[i][2] - 1;
            
            for(int j = i; j < k; j++){
                int ef = panels[j][0];
                int ey = panels[j][1] - 1;
                int ex = panels[j][2] - 1;
                
                if(i == j){
                    adjacencyMatrix[i][j] = 0;
                } else{
                    // 1. i -> elevator, 2. elevator 층 이동, 3. elevator -> j
                    int totalDist;
                    if(ef == sf){
                        totalDist = caculateDist(grid, sy, sx)[ey][ex];
                    }else{
                        totalDist = dist[sy][sx] + Math.abs(ef - sf) + dist[ey][ex];   
                    }
                    adjacencyMatrix[i][j] = totalDist;
                    adjacencyMatrix[j][i] = totalDist;
                }
            }
        }   
    }
    
// indegrees를 잘못 사용한 예시

    void makeRequiredMasks(int[][] seqs) {
        requiredMask = new int[k];
        for (int[] seq : seqs) {
            int prev = seq[0] - 1;
            int next = seq[1] - 1;
            // next를 열기 위해서는 prev 비트가 켜져 있어야 함
            requiredMask[next] |= (1 << prev);
        }
    }
    
    boolean inRange(int y, int x){
        return y < n && x < m && y >= 0 && x >= 0;
    }
    
    int tsp(int now, int visited){
        if(visited == ((1 << k) - 1)){
            return 0;
        }
        
        // 이미 방문한 적이 있으면 다시 방문하지 않음
        if(dp[now][visited] != -1) return dp[now][visited];
        dp[now][visited] = INF;
        
        for(int next = 0; next < k; next++){
            if((visited & (1 << next)) != 0) continue; // 이미 방문했으면 방문 X
            if((visited & requiredMask[next]) != requiredMask[next]) continue;
            
            int weight = adjacencyMatrix[now][next];
            dp[now][visited] = Math.min(
                dp[now][visited], 
                tsp(next, visited | (1 << next)) + weight
            );
        }
        
        return dp[now][visited];
    }
}
```

## 틀린 이유
1. 처음에는 indegrees 배열을 전역으로 두고, 하나씩 빼고 더하면서(직접 수정하면서) 0일 때 방문하는 방식을 사용
	1. requiredMask를 활용하여 위상 정렬을 하는 방법에 대해서도 알아두자. (indegree를 다시 복구해야 하는 상황에서 사용)
	2. DP 테이블 `dp[now][visited]`가 의미를 가지려면, now와 visited만으로 현재 상태가 유일하게 결정되어야 함
		1. 만약 외부 전역 변수(`indegrees`)가 재귀 호출 도중 계속 변한다면, 이전에 저장한 `dp` 값을 믿을 수 없게 됩니다.
2. `visited & (1 << next)`가 연산된 결과는 0 or 1이 아니다.
3. 같은 층에서 이동할 때는 엘리베이터를 이용하지 않아야 하지만, 이를 간과하고 항상 엘리베이터에서 출발했다.
4. tsp를 시작할 때, dp를 수정하지 않는다. weight는 tsp 함수 이후 더해주는 것일뿐. dp에 weight를 넣고 들어가지 않는다.
	1. TSP에서 `dp[i][mask]`는 보통 "현재 위치 `i`에 있고, `mask`만큼 방문했을 때 남은 거리"를 뜻합니다.
