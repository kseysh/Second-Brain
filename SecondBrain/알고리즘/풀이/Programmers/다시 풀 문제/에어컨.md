## 나의 풀이
```java
import java.util.*;

class Solution {
    class Pos{
        int time; 
        int temperature;
        Pos(int time, int temperature){
            this.time = time;
            this.temperature = temperature;
        }
    }
    
    private final int MAX_LEN = 1000;
    private final int MAX_TEMPERATURE = 50;
    private final int OFFSET = 10;
    private int[][] dp = new int[MAX_LEN + 1][MAX_TEMPERATURE + 1];
    private Queue<Pos> q = new LinkedList<>();
    
    private void initDp(){
        for(int i = 0; i <= MAX_LEN; i++){
            for(int t = 0; t <= MAX_TEMPERATURE; t++){
                dp[i][t] = Integer.MAX_VALUE;
            }
        }
    }
    
    private void simulate(int nextTime, int nextTemp, int nextCost, int[] onboard, int t1, int t2){
        if(nextTime >= onboard.length) return;
        if(nextTemp < 0 || nextTemp > 50) return;
        if(onboard[nextTime] == 1 && (nextTemp < t1 || nextTemp > t2)) return;
        
        if(dp[nextTime][nextTemp] > nextCost){
            dp[nextTime][nextTemp] = nextCost;
            q.add(new Pos(nextTime, nextTemp));
        }
    }
    
    public int solution(int temperature, int t1, int t2, int a, int b, int[] onboard) {
        boolean isWarmWeather = (t2 < temperature); // 더운 날씨인지
        
        initDp();
        t1 += OFFSET;
        t2 += OFFSET;
        temperature += OFFSET;
        
        q.add(new Pos(0, temperature));
        dp[0][temperature] = 0;
        while(!q.isEmpty()){
            Pos p = q.poll();
            int curTime = p.time;
            int curTemp = p.temperature;
            int curCost = dp[curTime][curTemp];
            int nextTime = curTime + 1;
            
            // 가만히 있어서 실내 온도를 떨어뜨린다.
            int nextCost = curCost;
            int nextTemp = curTemp + ((isWarmWeather) ? 1 : -1); // 더운 날씨에서는 가만히 있으면 실내 온도 올라감
            if(temperature == curTemp) nextTemp = temperature;
            simulate(nextTime, nextTemp, nextCost, onboard, t1, t2);

            // 현재 온도를 유지한다.
            nextCost = curCost + b;
            nextTemp = curTemp;
            simulate(nextTime, nextTemp, nextCost, onboard, t1, t2);
            
            // 현재 온도를 내리거나 올린다.
            nextCost = curCost + a;
            nextTemp = curTemp + ((isWarmWeather) ? -1 : 1);
            simulate(nextTime, nextTemp, nextCost, onboard, t1, t2);
        }
                
        int answer = Integer.MAX_VALUE;
        for(int t = 0; t <= 50; t++){
            answer = Math.min(answer, dp[onboard.length - 1][t]);
        }
        
        return answer;
    }
}
```
일단 푸는 시간이 오래걸림, DP임을 눈치채는 시간이 오래걸렸음
또한, 조건을 잘못걸어서 런타임 오류 발생 `(nextTemp < 0 && nextTemp > 50)` 이렇게 걸어서 true가 될 수 없는 조건을 사용함

## 다른 사람의 풀이
```java
class Solution {
    public int solution(int temperature, int t1, int t2, int a, int b, int[] onboard) {
        int[][] dp = new int[51][onboard.length+1];
        temperature += 10; t1 += 10; t2 += 10;

        for(int i = onboard.length-1; i >= 0; i--) {
            for(int t = 0; t < 51; t++) {
                dp[t][i] = Integer.MAX_VALUE - Math.max(a, b);
                if(onboard[i] == 1 && (t < t1 || t > t2)) continue;

                if(t == temperature) dp[t][i] = Math.min(dp[t][i], dp[t][i+1]);
                else if(t > 0 && t > temperature) dp[t][i] = Math.min(dp[t][i], dp[t-1][i+1]);
                else if(t < 50 && t < temperature) dp[t][i] = Math.min(dp[t][i], dp[t+1][i+1]);

                if(t < 50) dp[t][i] = Math.min(dp[t][i], dp[t+1][i+1]+a);
                if(t > 0) dp[t][i] = Math.min(dp[t][i], dp[t-1][i+1]+a);
                dp[t][i] = Math.min(dp[t][i], dp[t][i+1]+b);
            }
        }

        return dp[temperature][0];
    }
}
```
